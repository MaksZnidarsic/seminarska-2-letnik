\section{Uvod}


Pokažemo pristop k konstrukciji \textit{prve homotopske grupe} skozi primer matematičnega modeliranja problema obešanja slike (ang. picture hanging puzzle). Definicije in splošne izreke algebraične topologije povzamemo iz \cite{topologie-algebrique}, formulacija in rešitev problema pa sta predstavljeni v \cite{picture-hanging-puzzles}.

V nadaljevanju privzamemo naslednje splošne oznake: z $\Z$ označujemo množico celih in z $\R$ množico realnih števil, z $\I$ označujemo enotski interval, s $\S^n$ $n$-sfero in s $\T^n$ $n$-torus. Poleg tega s $\F_n$ označimo prosto grupo reda $n$.


\section{Obesiti sliko}


Problem si zastavimo takole: imamo sliko pritrjeno na sklenjeno vrvico, nekaj žebljev in prazno steno. Sliko z uporabo vrvice in žebljev pritrdimo na steno, kot kaže slika \ref{mucka}. Izziv je vrvico napeljati, tako da, ko iz stene odstranimo kateregakoli izmed žebljev, slika pade.

\begin{figure}[H]
    \centering
    \input{img/mucka}
    \caption{Primer obešene slike. Podana shema ne reši problema.}
    \label{mucka}
\end{figure}

Tukaj omenimo, da je to najpreprostejša oblika problema predstavljenega v \cite{picture-hanging-puzzles}. Tega posplošimo sledeče: denimo, da imamo v steni zabitih $n$ žebljev, tedaj zahtevamo, da slika pade natanko tedaj, ko iz nje odstranimo točno $k$ žebljev. Vendar, ker mi omenjeni problem uporabimo le v ilustrativne namene, se tukaj držimo enostavnejše verzije.


\section{Poti v prostoru}


Pričnemo z modeliranjem vrvi. Ker bi steno radi opisali z evklidsko ravnino, se zdi najbolj naravna izbira za model vrvi kar pot v prostoru.


\begin{definicija}
    Naj bo $X$ topološki prostor. Zvezna preslikava $\alpha : \I \to X$ je \textit{pot} v $X$. Element $x = \alpha(0)$ imenujemo \textit{začetna točka}, element $y = \alpha(1)$ pa \textit{končna točka} poti $\alpha$. Tedaj pravimo, da je $\alpha$ pot od $x$ do $y$.
\end{definicija}


\begin{definicija}
    Naj bosta $\alpha$ in $\beta$ poti v $X$, za kateri velja $\alpha(1) = \beta(0)$. Z $\alpha\beta$ označimo preslikavo $\I \to X$ podano s predpisom
    \begin{align*}
        x \mapsto
        \begin{cases}
            \alpha(2x) & (x \in [0, 1/2]) \\
            \beta(2x - 1) & (x \in [1/2, 1])
        \end{cases},
    \end{align*}
    ki ji pravimo \textit{stik poti} $\alpha$ in $\beta$.
\end{definicija}


\begin{trditev}
    Naj bodo $x, y, z \in X$ in $\alpha$ od $x$ do $y$ in $\beta$ od $y$ do $z$ poti v $X$. Tedaj je $\alpha\beta$ pot v $X$ od $x$ do $z$.
    \label{prop:produkt-poti-pot}
\end{trditev}


\begin{proof}
    Preprost z uporabo elementarnih izrekov s področja splošne topologije.
\end{proof}


\begin{definicija}
    Naj bo $\alpha$ pot v $X$. Z $\alpha^{-1}$ označimo \textit{obratno pot} poti $\alpha$ definirano s preslikavo $\I \to X, x \mapsto \alpha(1 - x)$.
\end{definicija}


\begin{opomba}
    Obratno pot se pogosto označuje $\overline\alpha$, vendar jo mi označujemo kot jo v duhu definicij in izrekov v sledečih poglavjih.
\end{opomba}


\section{Homotopija}


\begin{definicija}
    Naj bosta $X$ in $Y$ topološka prostora in $\phi, \psi : X \to Y$ zvezni. Preslikavi $\phi$ in $\psi$ sta \textit{homotopni} natanko tedaj, ko obstaja zvezna preslikava $h : X \times \I \to Y$, za katero velja, da je $h(x, 0) = \phi(x)$ in $h(x, 1) = \psi(x)$ za vse $x \in X$. Tedaj pišemo $\phi \simeq \psi$, preslikavi $h$ pa pravimo \textit{homotopija} med $\phi$ in $\psi$.

    Naj bo $A \subseteq X$. Če obstaja kakšna homotopija $h$, za katero velja $h(x, t) = \phi(x) = \psi(x)$ za vse $x \in A$, pravimo, da sta $\phi$ in $\psi$ \textit{homotopni relativno na} $A$. Pišemo $\phi \simeq \psi$ $(\text{rel } A)$.
\end{definicija}


\begin{trditev}
    Naj bosta $\phi, \psi : X \to Y$ zvezni. Preslikavi $\phi$ in $\psi$ sta homotopni natanko tedaj, ko obstaja pot med $\phi$ in $\psi$ v $\C(X, Y)$.
    \label{prop:homotopija-pot-v-zveznih-funkcijah}
\end{trditev}


\begin{proof}
    Sledimo dokazu podanem v \cite{topologie-algebrique}. Prepuščen bralcu.
\end{proof}


\begin{trditev}
    Homotopnost je ekvivalenčna relacija.
\end{trditev}


\begin{proof}
    Naj bodo $\phi, \psi, \sigma : X \to Y$ zvezne. Očitno velja $\phi \simeq \phi$. Naj bo $h$ homotopija med $\phi$ in $\psi$. Tedaj je $(x, y) \mapsto h(x, 1 - y)$ homotopija med $\psi$ in $\phi$. Nazadnje naj bo $\alpha$ pot od $\phi$ do $\psi$ in $\beta$ pot od $\psi$ do $\sigma$ v $\C(X, Y)$, potem je $\alpha\beta$ po trditvi \ref{prop:produkt-poti-pot} pot od $\phi$ do $\sigma$, torej sta homotopni po trditvi \ref{prop:homotopija-pot-v-zveznih-funkcijah}.
\end{proof}


\begin{definicija}
    Naj bosta $x, y \in X$ in $\alpha, \beta$ poti od $x$ do $y$. Poti $\alpha$ in $\beta$ sta \textit{homotopni glede na krajišča} natanko tedaj, ko sta homotopni relativno na $\{0, 1\}$.
\end{definicija}


\begin{opomba}
    Od tu naprej vedno, ko govorimo o homotopnosti poti, mislimo homotopnost glede na krajišča le-teh. Kljub temu pa je pomembno poudariti, da se moramo tega na vsakem koraku zavedati, saj velika večina izrekov in trditev temelji na tej predpostavki.
\end{opomba}


\section{Fundamentalna grupa}


\begin{definicija}
    Naj bo $X$ topološki prostor in $x \in X$. \textit{Zanka} v $x$ je pot v $X$ od $x$ do $x$. Z $Z(X, x)$ označimo \textit{množico vseh zank v} $x$.
\end{definicija}


\begin{lema} Naj bo $x \in X$ in $\alpha, \beta, \gamma$ zanke v $x$.
    \begin{enumerate}[(i)]
        \item $(\alpha\beta)\gamma \simeq \alpha(\beta\gamma)$
        \item $\alpha x \simeq x \alpha \simeq \alpha$,
        \item $\alpha^{-1}\alpha \simeq \alpha\alpha^{-1} \simeq x$.
    \end{enumerate}
    \label{lastnosti-grupe}
\end{lema}


\begin{proof}
    Naj bodo $\alpha, \beta, \gamma$ zanke v $x$.
\end{proof}


\begin{lema}
    Naj bodo $\alpha, \beta, \gamma, \delta$ poti v $X$, za katere velja $\alpha \simeq \gamma$ in $\beta \simeq \delta$. Če $\alpha\beta$ in $\gamma\delta$ obstajata, potem sta homotopna.
    \label{stik-poti-homotopen}
\end{lema}


\begin{proof}
    Naj bo $h$ homotopija med $\alpha$ in $\gamma$ ter $h'$ homotopija med $\beta$ in $\delta$. Lahko je preveriti, da je preslikava definirana s predpisom
    \begin{align*}
        (x, y) \mapsto
        \begin{cases}
            h(2x, y) & (x \in [0, 1/2]) \\
            h'(2x - 1, y) & (x \in [1/2, 1])
        \end{cases}
    \end{align*}
    želena homotopija.
\end{proof}


\begin{trditev}
    Naj bo $X$ topološki prostor in $x \in X$. Množica $Z(X, x)/\!\!\simeq$ opremljena z operacijo $([\alpha], [\beta]) \mapsto [\alpha\beta]$ tvori grupo.
\end{trditev}


\begin{proof}
    Sledi trivialno po lemah \ref{lastnosti-grupe} in \ref{stik-poti-homotopen}.
\end{proof}


\begin{definicija}
    Naj bo $X$ topološki prostor in $x \in X$. Grupi $Z(X, x) / \!\!\simeq$ pravimo \textit{prva homotopska} ali \textit{fundamentalna grupa} topološkega prostora $X$ in jo označimo $\pi_1(X, x)$.
\end{definicija}


\begin{zgled}
    Fundamentalna grupa $\S^1$ je trivialna.
\end{zgled}


\begin{zgled}
    Poglejmo si, kaj je fundamentalna grupa mnogoterosti $\T^2$. Naj bo $x \in \T^2$. Zgleda, kot da je $\pi_1(\T^2, x)$ neskončna in ima natanko dva generatorja — zanko, ki gre po obodu luknje, in zanko, ki obkroži biskvit —, zato bo potrebno preveriti le, ali je komutativna tj. ali je izomorfna $\Z^2$ ali $\F_2$.

    Naj bosta $\alpha, \beta$ generatorja $\pi_1(\T^2, x)$. Če si torus predstavljamo kot kvocient $\I^2$, je lahko videti, da velja $\alpha\beta\alpha^{-1}\beta^{-1} \simeq x$. Potem iz algebre 2 vemo, da mora biti $\pi_1(\T^2, x)$ komutativna. Posledično velja $\pi_1(\T^2, x) \cong \Z^2$.
\end{zgled}


\section{Obesimo sliko}


\begin{zgled}
    Poglejmo si model naše stene z uporabo evklidske ravnine. Pričnimo s prazno steno. Izberemo si točko $x \in \R^2$ — ta bo predstavljala našo sliko. Kot model vseh možnih načinov, na katere lahko na steno postavimo vrv, nam bo služila $Z(\R^2, x)$. Ker pa nam je vseeno, kako jo postavimo, saj se bo ta tako ali napela v trenutko, ko spustimo sliko, bo primernejša struktura kar $\pi_1(\R^2, x)$.

    Za model žebljev bomo uporabili luknje v ravnini, saj zanke, ki potujejo okoli ene izmed njih, niso homotopne $x$, torej slika ne bo padla. Naj bo $S \subset \R^2 \setminus \{x\}$ množica z $n$ elementi. Ni se težko prepričati, da izbira elementov $S$ nima vpliva na model. Seveda pa nas najbolj zanima $G = \pi_1(\R^2 \setminus S, x)$, ki modelira vse načine, na katere lahko obesimo sliko.

    Grupa $G$ bo za vsak $s \in S$ vsebovala ekvivalenčni razred vseh poti, ki gredo okoli $s$ in ne obrokožijo nobenega drugega elementa. Seveda pa sta taka ekvivalenčna razreda dva tj. naj bo $\alpha$ zanka kot zahtevano, potem sta $[\alpha]$ in $[\alpha^{-1}]$ iskana razreda. S preprostostjo v mislih ta dva elementa označujemo kar $s$ in $s^{-1}$. Lahko je videti, da je vsak element v $G$ možno napisati kot produkt teh osnovnih komponent. Sledi $G = \{\prod_i a_i : a_i \in \bigcup_{s \in S} \{s, s^{-1} \} \}$. Intuitivno smo vsako možno navijanje vrvi zapisali s koraki, ki smo jih morali narediti, da smo ga ustvarili. Kot smo pokazali, je možno $G$ zapisati kot množico vseh besed v $S$. Posledično velja, da je $G \cong \F_n$.

    Preostane nam še opis odstranjevanja enega izmed žebljev. Naj bo $\alpha$ pot, ki potuje okoli točke $s \in S$. Odstraniti žebelj pomeni odstraniti oviro — v našem primeru to pomeni zapolniti $s$, nakar bo $\alpha \simeq x$. Seveda pa ni nujno, da bo $\alpha$ tako preprosta, še vedno pa želimo, da je po transformaciji prostora homotopna $x$, saj to predstavlja padec slike. Torej za $s \in \F_n$ definirajmo $\phi_s : \F_n \to \F_n$ s predpisom
    \begin{align*}
        t \mapsto
        \begin{cases}
            e & (t = s) \\
            t & (t \neq s)
        \end{cases},
    \end{align*}
    kjer je $e$ identiteta in $t$ eden izmed generatorjev $\F_n$. Tako se vsa navijanja, ki padajo, ko odstranimo žebelj $s$, nahajajo v grupi $\ker \phi_s$. Torej rešitev našega problema leži v $\left(\bigcap_{s \in S} \ker \phi_s\right) \setminus \{e\}$.
    \label{konstrukcija-stene}
\end{zgled}


%Toliko časa in truda pa nam je vendarle uspelo; našli smo zelo preprost opis našega problema. To samo pokaže, česa vsega je človek zmožen, če si le dovoli razmišljati.


\begin{izrek}
    Naj bo $S$ množica generatorjev $\F_n$ in preslikave $\phi_s$ kot definirane v zgledu \ref{konstrukcija-stene}. Podgrupa $\bigcap_{s \in S} \ker \phi_s$ je netrivialna.
\end{izrek}


\begin{proof}
    Pričnimo s primerom za $n = 2$. Naj bosta $x, y$ generatorja $\F_2$. Označimo komutator $[x, y] = xyx^{-1}y^{-1}$. Ker $\F_n$ ni komutativna, velja $[x, y] \neq e$. Opazimo, da sta $\phi_x([x, y]), \phi_y([x, y]) = e$, torej je $\ker \phi_x \cap \ker \phi_y$ netrivialna.

    V splošnem naj bodo $x_i$ generatorji $\F_n$. Induktivno definiramo $S_1 = x_1$ in $S_n = [S_{n - 1}, x_n]$. Nadaljujemo z matematično indukcijo. Denimo, da je $S_p \in \ker \phi_i$ za vsak $i \in \{1, \dots, p\}$. Sledi, da je $\phi_{p + 1}(S_{p + 1}) = S_p S_p^{-1} = e$ in $\phi_i(S_{p + 1}) = x_{p + 1}x_{p + 1}^{-1} = e$ za $i \leq p$. Torej je $S_{p + 1} \in \bigcap_{i \leq p} \ker \phi_i$, kot želeno. Po indukciji sledi, da je $S_n \in \bigcap_{s \in S} \ker \phi_s$ kot zahtevano.
\end{proof}
